<?xml version="1.0" encoding="UTF-8"?>
<configuration scan="true" scanPeriod="60 seconds" debug="true">
<property name="log.lever" value="debug"/>
<property name="projectName" value="SoymilkCloud"/>
<property name="contextName" value="cloud-autosell"/>
<property name="logPath" value="/logs"/>
<contextName>${contextName}</contextName>
<property name="LOG_HOME" value="${logPath}/${projectName}/${contextName}/%d{yyyy-MM-dd}"/>
<property name="LOG_PATTERN"
          value="%date{yyyy-MM-dd HH:mm:ss} | %-5level | [%thread] | %logger{50} | %msg%n"/>
    <appender name="DEBUG" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!-- 按天回滚 daily -->
             <!-- 注意日期后需要加.%i -->
            <FileNamePattern>${LOG_HOME}/${contextName}-debug.%d{yyyy-MM-dd}.%i.log</FileNamePattern>
            <!-- 日志最大的历史 30天 -->
            <MaxHistory>30</MaxHistory>
            <cleanHistoryOnStart>true</cleanHistoryOnStart>
            <totalSizeCap>1GB</totalSizeCap>
            <MaxFileSize>100MB</MaxFileSize>
        </rollingPolicy>
        <encoder>
            <pattern>${LOG_PATTERN}</pattern>
            <charset class="java.nio.charset.Charset">UTF-8</charset>
        </encoder>
        <filter class="ch.qos.logback.classic.filter.LevelFilter"><!-- 只打印错误日志 -->
            <!--            <level>DEBUG</level>-->
            <onMatch>ACCEPT</onMatch>
            <onMismatch>DENY</onMismatch>
        </filter>
    </appender>
<!--     info日志 appender  -->
<appender name="INFO" class="ch.qos.logback.core.rolling.RollingFileAppender">
    <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
        <!-- 按天回滚 daily -->
        <FileNamePattern>${LOG_HOME}/${contextName}-info.%d{yyyy-MM-dd}.%i.log</FileNamePattern>
        <!-- 日志最大的历史 30天 -->
        <MaxHistory>30</MaxHistory>
        <cleanHistoryOnStart>true</cleanHistoryOnStart>
        <totalSizeCap>1GB</totalSizeCap>
        <MaxFileSize>100MB</MaxFileSize>
    </rollingPolicy>
    <encoder>
        <pattern>${LOG_PATTERN}</pattern>
        <charset class="java.nio.charset.Charset">UTF-8</charset>
    </encoder>
    <!-- 日志级别过滤器 -->
    <filter class="ch.qos.logback.core.filter.EvaluatorFilter">
        <evaluator>
            <expression>return message.contains("Preparing") || message.contains("Parameters:") || message.contains("Total:");</expression>
        </evaluator>
        <!--表示该级别及以下的，由下一个filter处理，如果当前是最后一个，则不匹配该级别以下的-->
        <OnMismatch>NEUTRAL</OnMismatch>
        <!--表示匹配该级别及以上-->
        <OnMatch>ACCEPT</OnMatch>
    </filter>

    <filter class="ch.qos.logback.classic.filter.ThresholdFilter">
        <!--过滤级别-->
        <level>INFO</level>
    </filter>
</appender>
    <appender name="WARN" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!-- 按天回滚 daily -->
            <FileNamePattern>${LOG_HOME}/${contextName}-warn.%d{yyyy-MM-dd}.%i.log</FileNamePattern>
            <!-- 日志最大的历史 30天 -->
            <MaxHistory>30</MaxHistory>
            <cleanHistoryOnStart>true</cleanHistoryOnStart>
            <totalSizeCap>1GB</totalSizeCap>
            <MaxFileSize>10MB</MaxFileSize>
        </rollingPolicy>
        <encoder class="ch.qos.logback.classic.encoder.PatternLayoutEncoder">
            <pattern>${LOG_PATTERN}</pattern>
            <charset class="java.nio.charset.Charset">UTF-8</charset>
        </encoder>
        <filter class="ch.qos.logback.classic.filter.LevelFilter"><!-- 只打印错误日志 -->
            <level>WARN</level>
            <onMatch>ACCEPT</onMatch>
            <onMismatch>DENY</onMismatch>
        </filter>
    </appender>
    <appender name="ERROR" class="ch.qos.logback.core.rolling.RollingFileAppender">
        <rollingPolicy class="ch.qos.logback.core.rolling.SizeAndTimeBasedRollingPolicy">
            <!-- 按天回滚 daily -->
            <FileNamePattern>${LOG_HOME}/${contextName}-error.%d{yyyy-MM-dd}.%i.log</FileNamePattern>
            <!-- 日志最大的历史 30天 -->
            <MaxHistory>30</MaxHistory>
            <cleanHistoryOnStart>true</cleanHistoryOnStart>
            <totalSizeCap>1GB</totalSizeCap>
            <MaxFileSize>10MB</MaxFileSize>
        </rollingPolicy>
        <encoder class="ch.qos.logback.classic.encoder.PatternLayoutEncoder">
            <pattern>${LOG_PATTERN}</pattern>
            <charset class="java.nio.charset.Charset">UTF-8</charset>
        </encoder>
        <filter class="ch.qos.logback.classic.filter.LevelFilter"><!-- 只打印错误日志 -->
            <level>ERROR</level>
            <onMatch>ACCEPT</onMatch>
            <onMismatch>DENY</onMismatch>
        </filter>
    </appender>
  <!-- 控制台日志彩色配置 -->
<property name="CONSOLE_LOG_PATTERN"
          value="%boldGreen(%date{yyyy-MM-dd HH:mm:ss}) | %highlight(%-5level) | %red([%thread]) | %boldMagenta(%logger{50}) | %boldCyan(%msg%n)"/>
<appender name="STDOUT" class="ch.qos.logback.core.ConsoleAppender">
    <encoder>
        <pattern>${LOG_PATTERN}</pattern>
        <charset class="java.nio.charset.Charset">UTF-8</charset>
        <pattern>${CONSOLE_LOG_PATTERN}</pattern>
    </encoder>
</appender>
<!-- FrameworkServlet日志-->
<logger name="org.springframework" level="INFO"/>

    <!--        <appender-ref ref="DEBUG"/>-->
    <appender-ref ref="STDOUT"/>
    <!--        <appender-ref ref="ERROR"/>-->
    <!--        <appender-ref ref="WARN"/>-->
    <appender-ref ref="INFO"/>
</logger>
<root level="INFO" additivity="false">
    <appender-ref ref="STDOUT"/>
</root>
</configuration>

问题总结： 1. 需求：打印system.out.printline的内容

  a.首先pom文件增加如下jar包
   <dependency>
    <groupId>uk.org.lidalia</groupId>
    <artifactId>sysout-over-slf4j</artifactId>
    <version>1.0.2</version>
  </dependency>

 b.在启动类添加以下代码
 SysOutOverSLF4J.sendSystemOutAndErrToSLF4J();

2.需求：把sql语句输出到log文件里方便排查错误，但是输出的日志级别控制为info以上。 一开始想改变sql输出级别，经大量尝试均无效，查阅资料表明 mybatis的sql输出级别只能设置为debug，debug以上级别就无法输出。后发现logback有自定义过滤器类（ch.qos.logback.core.filter.EvaluatorFilter

1

``<expression>return message.contains("Preparing") || message.contains("Parameters:") || message.contains("Total:");</expression>``

设置好表达式后，还有很重要的一个配置，需要正确设置匹配规则，另外自定义过滤器EvaluatorFilter应放在输出级别过滤器ThresholdFilter之前，否则两个过滤器互相冲突无法输出日志

<!--表示该级别及以下的，由下一个filter处理，如果当前是最后一个，则不匹配该级别以下的-->
<OnMismatch>NEUTRAL</OnMismatch>
<!--表示匹配该级别及以上-->
<OnMatch>ACCEPT</OnMatch>

以下为两个配置项的三个属性值：

1
2
3
4
5
6

    onMatch="ACCEPT" 表示匹配该级别及以上
    onMatch="DENY" 表示不匹配该级别及以上
    onMatch="NEUTRAL" 表示该级别及以上的，由下一个filter处理，如果当前是最后一个，则表示匹配该级别及以上
    onMismatch="ACCEPT" 表示匹配该级别以下
    onMismatch="NEUTRAL" 表示该级别及以下的，由下一个filter处理，如果当前是最后一个，则不匹配该级别以下的
    onMismatch="DENY" 表示不匹配该级别以下的
